⚙️ One-Time Setup — Google Sheets Central Database
Complete once to enable multi-user centralized score storage. Takes ~10 minutes.
Step 1 — Create Google Sheet Go to sheets.google.com → New spreadsheet → Name it "TeamPulse Scores"
Row 1 headers (A to K): Timestamp | Week | Edition | Member | Score | Correct | Wrong | Timeout | Violations | SectionScores | WeekAttempted
Step 2 — Apps Script Extensions → Apps Script → Delete all code → Paste this: 
function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  sheet.appendRow([
    new Date().toLocaleString('en-IN'),
    data.week, data.edition, data.member,
    data.score, data.correct, data.wrong,
    data.timeout, data.violations,
    JSON.stringify(data.sectionScores || {}),
    data.weekAttempted
  ]);
  return ContentService
    .createTextOutput(JSON.stringify({status:'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}
function doGet(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = sheet.getDataRange().getValues();
  var headers = data[0];
  var rows = data.slice(1).map(function(row) {
    var obj = {};
    headers.forEach(function(h,i){ obj[h]=row[i]; });
    return obj;
  });
  return ContentService
    .createTextOutput(JSON.stringify({data:rows}))
    .setMimeType(ContentService.MimeType.JSON);
}
Save → Deploy → New Deployment → Web App → Execute as Me → Anyone → Deploy → Copy URL
